Install-WindowsFeature DHCP -IncludeManagementTools

Add-DhcpServerv4Scope -Name "Réseau local" -StartRange 192.168.0.20 -EndRange 192.168.0.100 -SubnetMask 255.255.255.0 -State Active

Add-DhcpServerv4Failover -ComputerName "192.168.0.1" -Name "SFO-SIN-Failover" -PartnerServer "192.168.0.2" -ServerRole Standby -ScopeId 192.168.0.0 -force

Add-DnsServerPrimaryZone -ZoneName "com" -ZoneFile "com.dns" -verbose -passthru;

 Add-DnsServerResourceRecordA -Name ipssi -ZoneName com -IPv4Address 192.168.0.2


Set-DhcpServerv4OptionValue -DnsServer 192.168.0.2


Get-DnsServerResourceRecord -ZoneName com -RRType A

Add-DnsServerSecondaryZone -MasterServers 192.168.0.1 -Name com -ZoneFile com.dns


Set-DnsServerPrimaryZone -Name “com” –Notify Notifyservers –notifyservers “192.168.0.2” -SecondaryServers “192.168.0.2” –SecureSecondaries TransferToSecureServers


start-dnsserverzonesytransfer -Name com
